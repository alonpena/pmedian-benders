% ---------- 1 ----------
\begin{frame}{1. Problema de decisión: $p$-medianas}
\textbf{Decisión:} abrir exactamente $p$ sitios de $M$ candidatos para atender $N$ clientes.
Cada cliente se asigna al sitio abierto más cercano.
\[
  \min_{S\subseteq\{1,\dots,M\},\ |S|=p}\ \sum_{i=1}^{N}\min_{j\in S} d_{ij}
\]
\begin{columns}
\begin{column}{0.58\textwidth}
\textbf{Lectura industrial:}
\begin{itemize}
  \item bodegas / centros de distribución;
  \item refugios o puntos de atención;
  \item \textit{k-medoids}: centros representativos.
\end{itemize}
\end{column}
\begin{column}{0.38\textwidth}
\begin{block}{Estructura clave}
Fijados los sitios abiertos, asignar clientes es trivial.\\[3pt]
Lo difícil es elegir los $p$ sitios.
\end{block}
\end{column}
\end{columns}
\vfill
\footnotesize Problema NP-hard (Kariv \& Hakimi, 1979). Paper base: Duran-Mateluna, Ales \& Elloumi (2023).
\end{frame}

% ---------- 2 ----------
\begin{frame}{2. Por qué una formulación directa no escala}
\textbf{F1 clásica:} variables de apertura $y_j$ y asignación $x_{ij}$.
\[
\min \sum_i\sum_j d_{ij}x_{ij}\quad
\sum_j y_j=p,\quad \sum_j x_{ij}=1,\quad x_{ij}\le y_j
\]
\begin{itemize}
  \item Tamaño: $N\times M$ variables $x_{ij}$ y $N\times M$ restricciones de enlace.
  \item Si $N=M=10^4$: orden $10^8$ variables de asignación.
  \item El objetivo correcto no basta: el \textbf{modelo} debe entregar buenas cotas y matriz manejable.
\end{itemize}
\begin{block}{Idea de performance}
No resolver F1 “a fuerza bruta”. Reescribir el problema para explotar: radios ordenados, matriz rala, separación por cliente y lazy constraints.
\end{block}
\end{frame}

% ---------- 3 ----------
\begin{frame}{3. Formulaciones F1--F4: qué cambia realmente}
\small
\begin{tabular}{lll}
\toprule
 & idea & rol en este proyecto \\
\midrule
\textbf{F1} & asignación $x_{ij}\le y_j$ & baseline F1 (monolithic) parcial \\
\textbf{F2} & radios $z_i^k$, objetivo telescópico & antecedente teórico \\
\textbf{F3} & radios con cobertura exacta $d_{ij}=D_i^k$ & base de Benders \\
\textbf{F4} & cortes sobre $\theta_i$ & no se carga completa; se usa vía lazy constraints \\
\bottomrule
\end{tabular}
\vfill
\textbf{Punto de curso:} mismo óptimo entero y misma relajación LP; cambia densidad/tamaño.
\vfill
\begin{itemize}
  \item F3 preserva fortaleza LP, pero con matriz más rala.
  \item F4 completa sería un \emph{monolithic model} denso; por eso no se carga entera.
  \item Solver C principal: \textbf{maestro Benders + cortes F4 como lazy constraints}. No hay F2/F3/F4 completas cargadas como monolithic models.
\end{itemize}
\end{frame}

% ---------- 4 ----------
\begin{frame}{4. Relajación lagrangiana: contraste útil}
Desde F1, relajo asignación $\sum_j x_{ij}=1$ con multiplicadores $\lambda_i$:
\[
  L(\lambda)=\sum_i\lambda_i+
  \min_{\sum_j y_j=p}\sum_j \beta_j(\lambda)y_j,
  \qquad
  \beta_j(\lambda)=\sum_i\min(0,d_{ij}-\lambda_i).
\]
\begin{itemize}
  \item Subproblema lagrangiano separa por \textbf{sitio}: abrir los $p$ menores $\beta_j$.
  \item Entrega cota inferior: $L(\lambda)\le \OPT$.
  \item Recuperación primal: asignar a sitio abierto más cercano $\Rightarrow$ cota superior.
\end{itemize}
\begin{block}{Por qué lo incluyo}
Muestra misma motivación del curso: relajar/acotar para explotar estructura. Benders usa otra partición: fija $y$ y separa por cliente.
\end{block}
\end{frame}

% ---------- 5 ----------
\begin{frame}{5. Benders: maestro pequeño + subproblemas por cliente}
\textbf{Maestro:}
\[
  \min\sum_i\theta_i \quad
  \sum_j y_j=p,\quad y_j\in\{0,1\},\quad \theta_i\ge0,
\]
con cortes de Benders agregados iterativamente.
\vfill
\textbf{Subproblema $SP_i(\bar y)$:} dado $\bar y$, calcular distancia/costo del cliente $i$.
\begin{columns}
\begin{column}{0.52\textwidth}
\begin{itemize}
  \item Fijar $y$ separa F3 en $N$ subproblemas independientes.
  \item Las variables $z_i^k$ desaparecen del maestro.
  \item $\theta_i$ aproxima el costo real del cliente.
\end{itemize}
\end{column}
\begin{column}{0.44\textwidth}
\begin{block}{Sin feasibility cuts}
$SP_i$ siempre es factible y acotado: no hay capacidades ni asignaciones prohibidas, y $p\ge1$.\\
Solo hay optimality cuts.
\end{block}
\end{column}
\end{columns}
\end{frame}

% ---------- 6 ----------
\begin{frame}{6. Corte cerrado y Algoritmo 2 ($O(NM)$)}
Para cada cliente, ordenar sitios por distancia: $S_i=(S_{i1},\dots,S_{iM})$.
\[
  \kt_i=\text{último radio no cubierto por } \bar y.
\]
\vspace{-2mm}
\[
  \theta_i \ge D_i^{\kt_i+1}
  -\sum_{j:\ d_{ij}\le D_i^{\kt_i}}(D_i^{\kt_i+1}-d_{ij})y_j
\]
\begin{columns}
\begin{column}{0.55\textwidth}
\textbf{Algoritmo 2:}
\begin{itemize}
  \item recorre $S_i$ del sitio más cercano al más lejano;
  \item acumula masa $\bar y_j$ hasta cubrir 1;
  \item cuenta saltos estrictos de radio para obtener $\kt_i$.
\end{itemize}
\end{column}
\begin{column}{0.40\textwidth}
\begin{block}{Performance}
No resuelvo LPs de subproblema.\\
$O(M)$ por cliente, $O(NM)$ por separación.
\end{block}
\end{column}
\end{columns}
\vfill
\footnotesize Esta es la mejora central del paper: Benders, pero con dual cerrado.
\end{frame}
