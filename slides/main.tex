\begin{frame}{1. Problema y motivaciĂ³n}

\begin{frame}{2. Dificultad computacional}
\begin{itemize}
  \item \textbf{NP-duro} (Kariv \& Hakimi, 1979): elegir $p$ de $M$ son $\binom{M}{p}$ opciones.
  \item FormulaciĂ³n clĂ¡sica \textbf{F1} tiene $N\times M$ variables de asignaciĂ³n $x_{ij}$ y
        $\sim N\times M$ restricciones.
  \item Para $N=M=10^4 \Rightarrow \sim 10^8$ variables: los solvers genĂ©ricos colapsan.
\end{itemize}
\begin{block}{Idea para escalar}
No atacar el problema entero de frente, sino \textbf{explotar su estructura}: si fijamos quĂ©
sitios abrir, el problema se separa por cliente y se vuelve trivial.
\end{block}
$\Rightarrow$ esa observaciĂ³n es la semilla de la \textbf{relajaciĂ³n lagrangiana} y de
\textbf{Benders}.
\end{frame}

\begin{frame}{4. RelajaciĂ³n lagrangiana (propuesta teĂ³rica)}
Relajamos la \textbf{asignaciĂ³n} $\sum_j x_{ij}=1$ (restricciĂ³n complicante) con multiplicadores
$\lambda_i$:
\[
  L(\lambda)=\sum_i\lambda_i+\min_{\sum_j y_j=p}\sum_j \beta_j(\lambda)\,y_j,
  \quad \beta_j(\lambda)=\sum_i \min(0,\ d_{ij}-\lambda_i).
\]
\begin{itemize}
  \item El subproblema se separa \textbf{por sitio}: abrir los $p$ sitios con $\beta_j$ mĂ¡s
        negativo (Â¡por inspecciĂ³n, sin solver!).
  \item Cota inferior $L(\lambda)\le \OPT$; dual lagrangiano $\max_\lambda L(\lambda)$ vĂ­a
        \textbf{subgradiente} $g_i=1-\sum_j x_{ij}$.
  \item RecuperaciĂ³n primal: asignar cada cliente a su sitio abierto mĂ¡s cercano $\Rightarrow UB$.
\end{itemize}
\begin{block}{Contraste con Benders}
Lagrangiana separa por \textbf{sitio}; Benders fija $y$ y separa por \textbf{cliente}. Ambas dan
cota inferior.
\end{block}
\end{frame}

\begin{frame}{5. DescomposiciĂ³n de Benders}
\textbf{Maestro:} $\min\sum_i\theta_i$ s.a. $\sum_j y_j=p$, $y_j\in\{0,1\}$, + cortes.
\quad\textbf{Subproblema} $SP_i(\bar y)$: distancia del cliente $i$.
\[
  \theta_i \ge D^{\kt_i+1}_i-\!\!\sum_{j:\,d_{ij}\le D^{\kt_i}_i}\!\!\big(D^{\kt_i+1}_i-d_{ij}\big)y_j
  \qquad\text{(corte de optimalidad)}
\]
\begin{columns}
\begin{column}{0.54\textwidth}
\begin{itemize}
  \item $SP_i$ \textbf{siempre factible y acotado} $\Rightarrow$ \textbf{solo cortes de
        optimalidad}, nunca de factibilidad.
  \item Dual con \textbf{forma cerrada}: el corte se separa en $O(NM)$ \emph{sin resolver ningĂºn
        LP} (Ă­ndice $\kt_i$, Alg.\ 2).
\end{itemize}
\end{column}
\begin{column}{0.44\textwidth}
\footnotesize
\textbf{Por quĂ© se separa por cliente:} fijado $\bar y$, F3 se parte en $N$ problemas
independientes. Las variables $z$ desaparecen del maestro; las reemplaza un $\theta_i$.
\end{column}
\end{columns}
\vfill
\footnotesize \textbf{Validez:} cada corte es cota inferior vĂ¡lida de $\theta_i$ (no excluye el
Ă³ptimo); hay finitos cortes $\Rightarrow$ el branch-and-cut termina exacto.
\end{frame}
